~~~
<?xml version="1.0" encoding="UTF-8"?>
<!--
1. 每天生成一个文件，归档文件保存30天
2.
-->
<configuration >

    <!-- 配置好的属性在下面可以${}引入,在最上层控制 -->
    <!--设置自定义pattern属性-->
    <property name="pattern" value="%d{HH:mm:ss.SSS} [%-5level] [%thread] [%logger] %msg%n"/>
    <!-- 输出日志位置 -->
    <!-- 输出日志文件名 -->
    <!-- 输出日志级别 -->
    <!-- log.home jvm启动参数注入 -->
    <!--<property name="LOG_HOME" value="${log.home:-.}/logs"/>-->
    <property name="LOG_HOME" value="logs"/>
    <property name="LOG_FILE_NAME" value="${log.file.name:-OneForAll}"/>
    <property name="LOG_LEVEL" value="${log.level:-INFO}"/>

    <!-- 是否输出控制台日志 YES|NO -->
    <property name="LOG_CONSOLE" value="${log.console:-NO}"/>
    <property name="CONSOLE_ENCODING" value="${console.encoding:-UTF-8}"/>

    <!--控制台输出日志-->
    <appender name="CONSOLE" class="ch.qos.logback.core.ConsoleAppender">
        <!--设置控制台输出日志的格式-->
        <encoder>
            <pattern>${pattern}</pattern>
        </encoder>
    </appender>

    <!--滚动记录日志文件：-->
    <appender name="FILE" class="ch.qos.logback.core.rolling.RollingFileAppender">
        <!--当天生成的日志文件名称：-->
        <!--<file>/log/log.out</file>-->
        <file>${LOG_HOME}/${LOG_FILE_NAME}.log</file>
        <!--根据时间来记录日志文件：-->
        <rollingPolicy class="ch.qos.logback.core.rolling.TimeBasedRollingPolicy">
            <!--归档日志文件的名称：-->
            <fileNamePattern>${LOG_FILE_NAME}-%d{yyyy-MM-dd}.log</fileNamePattern>
            <!--归档文件保存30天-->
            <maxHistory>30</maxHistory>
            <!--<maxFileSize>50MB</maxFileSize>-->
        </rollingPolicy>
        <triggeringPolicy class="ch.qos.logback.core.rolling.SizeBasedTriggeringPolicy">
            <maxFileSize>1MB</maxFileSize>
        </triggeringPolicy>
        <!--生成的日志信息格式-->
        <encoder>
            <pattern>${pattern}</pattern>
        </encoder>
    </appender>

    <!--根root logger-->
    <root level="${LOG_LEVEL}">
        <!--设置根logger的日志输出目的地-->
        <!-- 输出到文件 -->
        <appender-ref ref="FILE" />
        <!-- 输出到控制台 -->
        <appender-ref ref="CONSOLE" />
    </root>

</configuration>
~~~
